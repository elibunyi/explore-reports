CSS Custom Properties

Copy and paste these CSS variables into your project's root stylesheet to maintain brand consistency across all Leisure Pools digital properties.

/* =====================================================
   LEISURE POOLS — CSS CUSTOM PROPERTIES
   Apply to :root in your base stylesheet
   ===================================================== */

:root {
  /* Core Colors */
  --color-primary:      #0057B7;   /* Primary Blue — main brand color */
  --color-gold:         #FFB81C;   /* Gold Accent — energy and warmth */
  --color-sky:          #0095C8;   /* Sky Blue — secondary brand color */
  --color-dark:         #1A1A1A;   /* Dark Text — near-black */
  --color-white:        #FFFFFF;   /* Pure White */
  --color-bg-light:     #F5F8FC;   /* Light Background — page default */

  /* Typography */
  --font-primary:       'Roboto', sans-serif;
  --font-size-base:     16px;
  --line-height-base:   1.7;

  /* Spacing */
  --radius:             8px;
  --radius-lg:          12px;
  --shadow:             0 4px 20px rgba(0,0,0,0.10);
  --shadow-hover:       0 8px 32px rgba(0,87,183,0.18);
  --transition:         all 0.22s ease;
  --container-max:      1200px;
}
